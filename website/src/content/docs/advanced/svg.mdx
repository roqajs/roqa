Rift provides first-class support for SVG (Scalable Vector Graphics).

## Basic SVG Components

```jsx
import { defineComponent, cell, get, set } from 'rift-js';

function Icon() {
  return (
    <svg width="24" height="24" viewBox="0 0 24 24">
      <path d="M12 2L2 7l10 5 10-5-10-5zM2 17l10 5 10-5M2 12l10 5 10-5"/>
    </svg>
  );
}

defineComponent('my-icon', Icon);
```

## Reactive SVG

SVG attributes can be reactive just like HTML:

```jsx
function ReactiveCircle() {
  const radius = cell(50);
  const color = cell('blue');

  return (
    <div>
      <svg width="200" height="200">
        <circle
          cx="100"
          cy="100"
          r={get(radius)}
          fill={get(color)}
        />
      </svg>

      <input
        type="range"
        min="10"
        max="90"
        value={get(radius)}
        oninput={(e) => set(radius, parseInt(e.target.value))}
      />

      <select
        value={get(color)}
        onchange={(e) => set(color, e.target.value)}
      >
        <option value="blue">Blue</option>
        <option value="red">Red</option>
        <option value="green">Green</option>
      </select>
    </div>
  );
}

defineComponent('reactive-circle', ReactiveCircle);
```

## Dynamic SVG Lists

Use `<For>` to render dynamic SVG elements:

```jsx
function BarChart() {
  const data = cell([
    { label: 'A', value: 30 },
    { label: 'B', value: 60 },
    { label: 'C', value: 45 },
    { label: 'D', value: 80 },
  ]);

  return (
    <svg width="300" height="200">
      <For each={data}>
        {(item, index) => (
          <g transform={`translate(${index * 70 + 20}, 0)`}>
            <rect
              x="0"
              y={200 - item.value * 2}
              width="50"
              height={item.value * 2}
              fill="steelblue"
            />
            <text x="25" y="195" text-anchor="middle" font-size="12">
              {item.label}
            </text>
          </g>
        )}
      </For>
    </svg>
  );
}

defineComponent('bar-chart', BarChart);
```

## SVG Animations

Animate SVG with cells and timers:

```jsx
function PulsingCircle() {
  const scale = cell(1);
  let growing = true;

  this.connected(() => {
    const animate = () => {
      const current = get(scale);
      if (growing) {
        set(scale, current + 0.02);
        if (current >= 1.5) growing = false;
      } else {
        set(scale, current - 0.02);
        if (current <= 1) growing = true;
      }
    };

    const intervalId = setInterval(animate, 16);

    return () => {
      clearInterval(intervalId);
    };
  });

  return (
    <svg width="200" height="200" viewBox="0 0 200 200">
      <circle
        cx="100"
        cy="100"
        r="50"
        fill="coral"
        transform={`scale(${get(scale)})`}
        transform-origin="100 100"
      />
    </svg>
  );
}

defineComponent('pulsing-circle', PulsingCircle);
```

## Integrating with D3.js

Rift works well with D3.js for complex visualizations:

```jsx
import { defineComponent, cell, get, set, bind } from 'rift-js';
import * as d3 from 'd3';

function D3Chart() {
  const data = cell([10, 20, 30, 40, 50]);

  this.connected(() => {
    const svg = this.querySelector('svg');

    // Initial render
    renderChart(svg, get(data));

    // Re-render when data changes
    bind(data, (newData) => {
      renderChart(svg, newData);
    });
  });

  const renderChart = (svg, data) => {
    const selection = d3.select(svg)
      .selectAll('rect')
      .data(data);

    selection.enter()
      .append('rect')
      .merge(selection)
      .attr('x', (d, i) => i * 60)
      .attr('y', d => 200 - d * 3)
      .attr('width', 50)
      .attr('height', d => d * 3)
      .attr('fill', 'steelblue');

    selection.exit().remove();
  };

  const randomize = () => {
    set(data, [1, 2, 3, 4, 5].map(() => Math.floor(Math.random() * 60) + 10));
  };

  return (
    <div>
      <svg width="300" height="200"></svg>
      <button onclick={randomize}>Randomize</button>
    </div>
  );
}

defineComponent('d3-chart', D3Chart);
```

<Aside type="tip">
When using D3, let D3 handle the SVG manipulation and use Rift cells to manage the data. Use `bind()` to re-render when data changes.
</Aside>

## SVG Icons with Props

Create reusable icon components:

```jsx
function Icon(props) {
  const { name, size = 24, color = 'currentColor' } = props;

  const icons = {
    check: 'M20 6L9 17l-5-5',
    x: 'M18 6L6 18M6 6l12 12',
    arrow: 'M5 12h14m-7-7l7 7-7 7',
  };

  return (
    <svg
      width={size}
      height={size}
      viewBox="0 0 24 24"
      fill="none"
      stroke={color}
      stroke-width="2"
      stroke-linecap="round"
      stroke-linejoin="round"
    >
      <path d={icons[name] || icons.check}/>
    </svg>
  );
}

defineComponent('my-icon', Icon);
```

Usage:

```html
<my-icon name="check" size="32" color="green"></my-icon>
<my-icon name="x" size="16" color="red"></my-icon>
```

## Handling SVG Events

SVG elements support the same events as HTML:

```jsx
function InteractiveMap() {
  const hoveredRegion = cell(null);

  return (
    <div>
      <svg width="400" height="300" viewBox="0 0 400 300">
        <path
          d="M50,50 L150,50 L150,150 L50,150 Z"
          fill={get(hoveredRegion) === 'region1' ? 'lightblue' : 'gray'}
          onmouseover={() => set(hoveredRegion, 'region1')}
          onmouseout={() => set(hoveredRegion, null)}
        />
        <path
          d="M200,50 L300,50 L300,150 L200,150 Z"
          fill={get(hoveredRegion) === 'region2' ? 'lightgreen' : 'gray'}
          onmouseover={() => set(hoveredRegion, 'region2')}
          onmouseout={() => set(hoveredRegion, null)}
        />
      </svg>
      <p>Hovered: {get(hoveredRegion) || 'none'}</p>
    </div>
  );
}

defineComponent('interactive-map', InteractiveMap);
```

## SVG + CSS Animations

Combine SVG with CSS for smooth animations:

```jsx
function AnimatedLoader() {
  return (
    <svg class="loader" width="50" height="50" viewBox="0 0 50 50">
      <circle
        cx="25"
        cy="25"
        r="20"
        fill="none"
        stroke="currentColor"
        stroke-width="4"
        stroke-dasharray="80, 200"
        stroke-linecap="round"
      />
    </svg>
  );
}

defineComponent('animated-loader', AnimatedLoader);
```

```css
.loader {
  animation: rotate 1s linear infinite;
}

.loader circle {
  animation: dash 1.5s ease-in-out infinite;
}

@keyframes rotate {
  100% { transform: rotate(360deg); }
}

@keyframes dash {
  0% {
    stroke-dasharray: 1, 200;
    stroke-dashoffset: 0;
  }
  50% {
    stroke-dasharray: 80, 200;
    stroke-dashoffset: -35;
  }
  100% {
    stroke-dasharray: 80, 200;
    stroke-dashoffset: -124;
  }
}
```
